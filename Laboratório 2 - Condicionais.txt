{

    int delta = 0;
    
    //variable to help in the treatment of delta
    int result = 0;

    printf("Write c: ");

    if (a != 0) {
        //calculate delta (b² - 4ac)
        delta = (b*b) - (4*a*c);

            case 3:

        //calculate the equation with the delta that already is found
        if (result > 1) 
        {
            //The 2 roots of the equation
            x1 = (-b + sqrt(delta)) / 2*a;
            x2 = (-b - sqrt(delta)) / 2*a;

        }

    // If a is equal to zero

        printf("if a is equal to zero, so it's not a second degree equation.");
    }

*/

main()
{

    //Counter to use 'for' function

    //Result to count how many times the user got it right
    int result = 0;
    //to see the calculus

    //To help rand() funcition give random number (giving it a seed)
    srand(time(NULL));
    //A loop to ask the user 5 times
    for (counter = 1; counter <= 5; counter++)

        //Function to give random numbers to the user
        num1 = rand() % 100;
        num2 = rand() % 100;
        
        //To ask the user about the sum
        printf("What is the sum of: %d + %d?\n", num1, num2);
        //Get the users answer
        scanf("%d", &sum);
        
        //Test if the 
        if ((num1 + num2) == sum)
        {
            //To count how many times the user answer right
            result += 1;
            printf("You got it!\n");
        }
        else
        {
            //Say that the answer is wrong
            printf("You missed, try to pay attention next time!");
        }
        //Explain the sum
        printf("The result is %d + %d = %d\n\n", num1, num2, sum);
        
    }
